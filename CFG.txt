<START> -> <PROG_HEADER><BODY> | <PROG_HEADER><MAIN_BODY>

<PROG_HEADER> -> <PROG_DECL><PROG_DECL_REST>
<PROG_DECL> -> P
<PROG_DECL_REST> -> <PROG_NAME><SEMICOLON>
<PROG_NAME> -> Z

<BODY> -> <VAR><MAIN_BODY>
<IDENTIFIER> -> Z
<NUM> -> Z

<VAR> -> <VAR_HEADER><VAR_BODY>
<VAR_HEADER> -> V
<VAR_BODY> -> <IDENTIFIER><TYPE_DECL> | <IDENTIFIER><MORE_VAR> | <VAR_BODY><VAR_BODY>
<MORE_VAR> -> <COMMA><VAR_BODY>
<COMMA> -> CM
<TYPE_DECL> -> <COLON><VAR_TYPE>
<COLON> -> C
<VAR_TYPE> -> <PRIV_TYPE><SEMICOLON>
<PRIV_TYPE> -> INT | RE | CH | <ARRAY><PRIV_TYPE>
<ARRAY> -> <ARRAY_MARKER><ARRAY_REST>
<ARRAY_MARKER> -> AR
<ARRAY_REST> -> <OPEN_BRACKET><RANGE_REST>
<OPEN_BRACKET> -> OB
<RANGE_REST> -> <NUM><RANGE_REST_2>
<RANGE_REST_2> -> <RANGE_SEPARATOR><NUM_END>
<RANGE_SEPARATOR> -> DP
<NUM_END> -> <NUM><ARRAY_CLOSER>
<ARRAY_CLOSER> -> <CLOSE_BRACKET><OF>
<CLOSE_BRACKET> -> CB
<OF> -> OF
<SEMICOLON> -> SC

<MAIN_BODY> -> <BEGIN><MAIN_PROGRAM>
<BEGIN> -> B
<MAIN_PROGRAM> -> <ASSIGNMENT><MAIN_PROGRAM> | <CONDITIONAL><MAIN_PROGRAM> | <WHILE_LOOP><MAIN_PROGRAM> | <FOR_LOOP><MAIN_PROGRAM> | <REPEAT_LOOP><MAIN_PROGRAM> | <INPUT_OUTPUT><MAIN_PROGRAM> | <END><PERIOD>
<END> -> E
<PERIOD> -> PR

<ASSIGNMENT> -> <IDENTIFIER><REST_ASSIGNMENT> | <ARRAY_ID><REST_ASSIGNMENT>
<ARRAY_ID> -> <IDENTIFIER><ARRAY_ELM>
<ARRAY_ELM> -> <OPEN_BRACKET><REST_ARRAY_ELM> | <ARRAY_ELM><ARRAY_ELM>
<REST_ARRAY_ELM> -> <NUM><CLOSE_BRACKET> | <ARRAY_ID><CLOSE_BRACKET> | <IDENTIFIER><CLOSE_BRACKET>
<REST_ASSIGNMENT> -> <EQUALS_TO><RIGHTHAND_ASSIGNMENT>
<EQUALS_TO> -> A
<RIGHTHAND_ASSIGNMENT> -> <IDENTIFIER><SEMICOLON> | <ARRAY_ID><SEMICOLON> | <EXPRESSION><SEMICOLON> | <PARENTHESIS_EXPR><SEMICOLON>
<EXPRESSION> -> <IDENTIFIER><EXPR_REST> | <NUM><EXPR_REST> | <ARRAY_ID><EXPR_REST> | <PARENTHESIS_EXPR><EXPR_REST>
<EXPR_REST> -> <MATH_OPERATOR><RIGHT_EXPR> | <COND_OPERATOR><RIGHT_EXPR>
<PARENTHESIS_EXPR> -> <OPEN_PARENTHESIS><PAREN_EXPR_REST>
<PAREN_EXPR_REST> -> <EXPRESSION><CLOSE_PARENTHESIS>
<RIGHT_EXPR> -> Z | <IDENTIFIER><EXPR_REST> | <NUM><EXPR_REST> | <ARRAY_ID><EXPR_REST> | <PARENTHESIS_EXPR><EXPR_REST>
<MATH_OPERATOR> -> PL | SB | MP | DV | IDV | MD
<COND_OPERATOR> -> CMP

<CONDITIONAL> -> <CONDITIONAL_HEADER><MAIN_BODY_INSIDE>
<CONDITIONAL_HEADER> -> <FIRST_TYPE_CONDITIONAL><COND_HEADER_REST>
<FIRST_TYPE_CONDITIONAL> -> IF

<MAIN_BODY_INSIDE> -> <BEGIN><MAIN_PROGRAM_INSIDE>

<MAIN_PROGRAM_INSIDE> -> <ASSIGNMENT><MAIN_PROGRAM_INSIDE> | <CONDITIONAL><MAIN_PROGRAM_INSIDE> | <WHILE_LOOP><MAIN_PROGRAM_INSIDE> | <FOR_LOOP><MAIN_PROGRAM_INSIDE>  | <REPEAT_LOOP><MAIN_PROGRAM_INSIDE>  | <INPUT_OUTPUT><MAIN_PROGRAM_INSIDE> | <END><SEMICOLON> | <END><NEXT_CONDITIONAL>

<NEXT_CONDITIONAL> -> <ELSE><CONDITIONAL> | <ELSE><MAIN_BODY_INSIDE>
<ELSE> -> S
<COND_HEADER_REST> -> <COND_IF_EXPRESSION><THEN>
<THEN> -> TH
<COND_IF_EXPRESSION> -> <LEFT_COND_EXPRESSION><RIGHT_COND_EXPRESSION> | <COND_EXPRESSION><NEXT_COND> | <OPEN_PARENTHESIS><EXPRESSION_REST>
<OPEN_PARENTHESIS> -> OP

<EXPRESSION_REST> -> <COND_EXPRESSION><NEXT_COND> | <COND_EXPRESSION><CLOSE_PARENTHESIS>

<CLOSE_PARENTHESIS> -> CP
<NEXT_COND> ->  <LOGICAL_OPERATOR><EXPRESSION_REST>
<LOGICAL_OPERATOR>  -> AN | OR
<COND_EXPRESSION> -> <LEFT_COND_EXPRESSION><RIGHT_COND_EXPRESSION> | <OPEN_PARENTHESIS><PAREN_COND_EXPR_REST>
<LEFT_COND_EXPRESSION> -> Z | <IDENTIFIER><OPERATION> | <ARRAY_ID><OPERATION> | <OPEN_PARENTHESIS><PAREN_COND_EXPR>
<PAREN_COND_EXPR_REST> -> <COND_EXPRESSION><CLOSE_PARENTHESIS>
<PAREN_COND_EXPR> -> <LEFT_COND_EXPRESSION><CLOSE_PARENTHESIS>
<OPERATION> -> <MATH_OPERATOR><IDENTIFIER> | <MATH_OPERATOR><ARRAY_ID>
<RIGHT_COND_EXPRESSION> -> <COND_OPERATOR><LEFT_COND_EXPRESSION>

<WHILE_LOOP> -> <WHILE_EXPRESSION><LOOP_REST>
<WHILE_EXPRESSION> -> <WHILE><COND_IF_EXPRESSION>
<WHILE> -> W
<LOOP_REST> -> <DO><MAIN_BODY_LOOP>
<DO> -> D
<MAIN_BODY_LOOP> -> <BEGIN><MAIN_PROGRAM_LOOP>

<MAIN_PROGRAM_LOOP> -> <ASSIGNMENT><MAIN_PROGRAM_LOOP> | <CONDITIONAL><MAIN_PROGRAM_LOOP> | <WHILE_LOOP><MAIN_PROGRAM_LOOP> | <FOR_LOOP><MAIN_PROGRAM_LOOP>  | <REPEAT_LOOP><MAIN_PROGRAM_LOOP>  | <INPUT_OUTPUT><MAIN_PROGRAM_LOOP> | <END><SEMICOLON>

<FOR_LOOP> -> <FOR><FOR_HEADER>
<FOR_HEADER> -> <LEFT_FOR><FOR_REST>
<FOR> -> F
<LEFT_FOR> -> <IDENTIFIER><R_H_LEFT_FOR> | <ARRAY_ID><R_H_LEFT_FOR>
<R_H_LEFT_FOR> -> <EQUALS_TO><IDENTIFIER> | <EQUALS_TO><EXPRESSION> | <EQUALS_TO><ARRAY_ID>
<FOR_REST> -> <INC/DEC><RIGHT_FOR>
<INC/DEC> -> T | DT
<RIGHT_FOR> -> <IDENTIFIER><STEP_FOR> | <IDENTIFIER><LOOP_REST> | <EXPRESSION><STEP_FOR> | <EXPRESSION><LOOP_REST> | <ARRAY_ID><STEP_FOR> | <ARRAY_ID><LOOP_REST>
<STEP_FOR> -> <STEP_EXPRESSION><LOOP_REST>
<STEP_EXPRESSION> -> <STEP><IDENTIFIER> | <STEP><EXPRESSION> | <STEP><ARRAY_ID>
<STEP> -> ST

<REPEAT_LOOP> -> <REPEAT><MAIN_PROG_REPEAT>
<REPEAT> -> R
<MAIN_PROG_REPEAT> -> <ASSIGNMENT><MAIN_PROG_REPEAT> | <CONDITIONAL><MAIN_PROG_REPEAT> | <WHILE_LOOP><MAIN_PROG_REPEAT> | <FOR_LOOP><MAIN_PROG_REPEAT> | <REPEAT_LOOP><MAIN_PROG_REPEAT> | <INPUT_OUTPUT><MAIN_PROG_REPEAT> | <UNTIL><UNTIL_REST>
<UNTIL_REST> -> <COND_IF_EXPRESSION><SEMICOLON>
<UNTIL> -> U

<QUOTES> -> K
<INPUT_OUTPUT> -> <INPUT><PARAM_IN> | <OUTPUT><PARAM_OUT>
<INPUT> -> I
<OUTPUT> -> O
<PARAM_IN> -> <OPEN_PARENTHESIS><PARAM_IN_REST>
<PARAM_IN_REST> -> <IDENTIFIER><NEXT_IDENTIFIER_IN> | <ARRAY_ID><NEXT_IDENTIFIER_IN>
<NEXT_IDENTIFIER_IN> -> <COMMA><PARAM_IN_REST> | <CLOSE_PARENTHESIS><SEMICOLON>
<PARAM_OUT> -> <OPEN_PARENTHESIS><PARAM_OUT_REST>
<PARAM_OUT_REST> -> <IDENTIFIER><NEXT_IDENTIFIER_OUT> | <ARRAY_ID><NEXT_IDENTIFIER_OUT> | <EXPRESSION><NEXT_IDENTIFIER_OUT> | <TEXT><NEXT_IDENTIFIER_OUT>
<NEXT_IDENTIFIER_OUT> -> <COMMA><PARAM_OUT_REST> | <CLOSE_PARENTHESIS><SEMICOLON>
<TEXT> -> <QUOTES><TEXT_REST>
<TEXT_REST> -> <TEXT_BODY><ADD_TEXT>
<ADD_TEXT> -> K | <PLUS><TEXT>
<PLUS> -> PL
